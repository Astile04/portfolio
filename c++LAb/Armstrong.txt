#include<iostream>
using namespace std;

int main(){
    int x,arm,sum=0;
    int temp;
    
    cout<<"Enter a num:";
    cin>>x;
    
    temp=x;
    
    while(x!=0){
        arm=x%10;
        sum=sum+arm*arm*arm;
        x=x/10;
    }
    
    if(temp==sum){
        cout<<"Armstrong";
    }else{
        cout<<"Not armstrong";
    }
}
