#include<iostream>
#include<cmath>
using namespace std;


void swap(int x,int y){
	int  temp=x;
	x=y;
	y=temp;
	cout<<"After swapping:"<<x<<":"<<y;
}



int  main(){
	int x,y;
	
	cout<<"Enter two numbers:";
	cin>>x>>y;
	
	cout<<"Before swapping:"<<x<<" "<<y;
	
	swap(x,y);
	
	
}
