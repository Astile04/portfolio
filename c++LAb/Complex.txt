#include<iostream>
using namespace std;

class Complex_number{
	private:
		int  a,b,c,d;
		int real,imag;
		
	public:
		void get(){
			cout<<"Enter a and b wherea+ib:";
			cin>>a>>b;
			
			cout<<"Enter c and d where c+id:";
			cin>>c>>d;
		}
		
		
		void add(){
			real=a+c;
			imag=b+d;
			
			cout<<"Sum is:"<<real<<"+"<<imag<<"i";
		}
		
		void sub(){
			real=a-c;
			imag=b-d;
			
			cout<<"Difference is:"<<real<<"-"<<imag<<"i";
		}
};

int main(){
	Complex_number cn;
	cn.get();
	cn.add();
	cn.sub();
}
