#include<iostream>
#include<cmath>
using namespace std;

class Triangle{
	private:
		int a,b,c;
		float s,area;
		
	public:
		void get(){
			cout<<"Enter 3 sides:";
			cin>>a>>b>>c;
			
			if((a+b>c)&&(b+c>a)&&(a+c>b)){
				s=(a+b+c)/2;
				area=sqrt(s*(s-a)*(s-b)*(s-c));
				cout<<"Area is:"<<area;
			}else{
				cout<<"INvalid triangle";
			}
		}
};

int main(){
	Triangle t;
	t.get();
}
