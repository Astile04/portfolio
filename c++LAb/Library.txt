#include<iostream>
#include<cstring>
using namespace std;

class Library{
	private:
		int x;
		string books[100];
		char book_name[20];
		int book_counter;
	public:
		Library(){
			book_counter=0;
		}
		
		void add(){
			cout<<"Enter how many books you want to add:";
			cin>>x;
			cin.ignore();
			for(int i=0;i<x;i++){
				cout<<"Enter the book "<<" "<<i+1<<":";
				getline(cin,books[i]);
				book_counter++;
			}
		}
		
		
		void remove(){
			cout<<"Enter the book name to remove:";
			cin>>book_name;
			
			for(int i=0;i<book_counter;i++){
				if(book_name==books[i]){
					for(int j=i;j<book_counter;j++){
						books[j]=books[j+1];
						book_counter--;
					}
				}
			}
		}
		
		void display(){
			cout<<"Displaying the books:";
			for(int i=0;i<book_counter;i++){
				cout<<books[i];
			}
		}
};

int main(){
	Library l;
	l.add();
	l.remove();
	l.display();
}
