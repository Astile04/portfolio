#include<iostream>
#include<cstring>
using namespace std;

int main(){
    char word[20];
    cout<<"Enter the word;";
    cin>>word;
    
    int length=strlen(word);
    
    string str;
    
    for(int i=length-1;i>=0;i--){
        str+=word[i];
    }
    
    if(word==str){
        cout<<"Palindrome";
    }else{
        cout<<"Not palindrome";
    }
}
