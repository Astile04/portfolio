#include<iostream>
using namespace std;

class worker{
	private:
		char wname[25];
		float hrwrk,wgrate;
		float totwage;
		
		float calcwage(){
			return hrwrk*wgrate;
		}
		
	public:
		void in_data(){
			cout<<"Enter the worker name:";
			cin>>wname;
			
			cout<<"Enter the hrwrk:";
			cin>>hrwrk;
			
			cout<<"Enter the wgrate:";
			cin>>wgrate;
			
			totwage=calcwage();
		}
		
		
		void out_data(){
			cout<<"Worker name is:"<<wname;
			cout<<"Total wage is:"<<totwage;
		}
};

int main(){
	worker w;
	w.in_data();
	w.out_data();
}
