#include <iostream>
using namespace std;

class Weight {
private:
    int kg;

public:
    Weight() {
        kg = 0;
    }

    Weight(int x) {
        kg = x;
    }

    // Overloaded prefix ++ operator
    void operator++() {
        ++kg;
    }

    // Overloaded postfix ++ operator
    void operator++(int) {
        kg++;
    }

    void print() {
        cout << "Weight in kg is: " << kg << endl;
    }
};

int main() {
    Weight obj;

    obj.print(); // Initial value

    ++obj; // Prefix increment
    obj.print();

    obj++; // Postfix increment
    obj.print();

    return 0;
}

class Num{
	private:
		int x;
		
	public:
		Num(){
			x=0;
		}
		
		Num(int value){
			x=value;
		}
		
		void operator -(){
			x=-x;
		}
		
		
		void print(){
			cout<<"Value:"<<x;
		}
};

int main(){
	Num n(10);
	

	n.print();
	
	-n;
	n.print();
}
